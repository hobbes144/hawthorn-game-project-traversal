77/*!****************************************************************************
78 * \brief Remove a component
79 *
80 * ## Usage:
81 *
82 * This is to be used when a GameObject no longer needs a specific component.
83 * Note that this is a heavy call so it is to be avoided as much as possible.
84 * Prefer to use events to disable components.
85 *
86 * \param component Component to be removed.
87 *****************************************************************************/
88template<typename T>
89void GameObject::removeComponent() {
90 std::shared_ptr<T> found = nullptr;
91 auto it = std::remove_if(
92 components.begin(),
93 components.end(),
94 [&](const std::shared_ptr<Component>& c) {
95 found = std::dynamic_pointer_cast<T>(c);
96 return found != nullptr;
97 });
98 if (it != components.end()) {
99 found->shutdown();
100 components.erase(it);
101 }
102}
