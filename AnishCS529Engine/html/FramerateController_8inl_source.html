79/*!****************************************************************************
80 * \brief Runs the end of frame logic
81 *
82 * Uses a busy-wait loop to check if frame has ended.
83 * Also updates accumulators.
84 *
85 *****************************************************************************/
86template<typename T>
87void FramerateController<T>::endFrame() {
88 renderTime = static_cast<T>(glfwGetTime()) - timeAtFrameStart;
89 frameCount++; framesSinceLastFPSQuery++;
90 /*T remainingTime = targetFrameTime - renderTime;
91 if (remainingTime > 0) {
92 std::this_thread::sleep_for(
93 std::chrono::duration<T>(remainingTime * 0.9));
94 }*/
95 while ((static_cast<T>(glfwGetTime()) - timeAtFrameStart) < targetFrameTime);
96 timeAtFrameEnd = static_cast<T>(glfwGetTime());
97 lastFrameTime = timeAtFrameEnd - timeAtFrameStart;
98 physicsAccumulator += lastFrameTime;
99
100#ifdef ENABLE_ADDITIONAL_ACCUMULATORS
101 for (unsigned int i = 0; i < accumulatorsRegistered; i++)
102 accumulatorTimes[i] += lastFrameTime;
103#endif // ENABLE_ADDITIONAL_ACCUMULATORS
104
105}

107/*!****************************************************************************
108 * \brief Gets the FPS
109 *
110 * This is calculated as the average of frames created per second since the
111 * last getFPS() call.
112 *
113 * \return \b T Current frames per second.
114 *****************************************************************************/
115template<typename T>
116T FramerateController<T>::getFPS() {
117 T lastQuery = timeOfLastFPSQuery;
118 long int numFrames = framesSinceLastFPSQuery;
119 timeOfLastFPSQuery = static_cast<T>(glfwGetTime());
120 framesSinceLastFPSQuery = 0;
121 return (numFrames / (static_cast<T>(glfwGetTime()) - lastQuery));
122}

194/*!****************************************************************************
195 * \brief Creates a rate controller
196 *
197 * \b Note: This feature is only enabled when the flag
198 * ENABLE_RATE_CONTROLLERS is set
199 *
200 * Rate controllers are separate timers that are used to check if at least the
201 * designated rate has elapsed since last successful fire.
202 *
203 * This is low budget since it does not get updated every frame, and instead
204 * only calculates when asked.
205 *
206 * This can be expanded to have the functionality of an accumulator.
207 * Tradeoff: This method has longer calc time during checking, but it has
208 * lesser impact on endFrame().
209 *
210 * Possible direction: We could extend this to have a callback function that
211 * we can trigger in the endFrame() loop.
212 *
213 * \param rate
214 * \return \b
215 *****************************************************************************/
216template<typename T>
217const unsigned int FramerateController<T>::createRateController(
218 const unsigned int rate) {
219
220 unsigned int rateControllerId = rateControllersRegistered;
221 rateControllersRegistered++;
222 rateControllerTimes.push_back(static_cast<T>(glfwGetTime()));
223 rateControllerTimes[rateControllerId] = static_cast<T>(glfwGetTime());
224 rateControllerTargetRates.push_back(1.0 / rate);
225 rateControllerTargetRates[rateControllerId] = (1.0 / rate);
226
227 return rateControllerId;
228}

230/*!****************************************************************************
231 * \brief If the specified rate controller should fire
232 *
233 * Checks if at least `1/rate` time has elapsed since last fire.
234 * This also updates the last fire time.
235 *
236 * \param rateControllerId
237 * \return \b
238 *****************************************************************************/
239template<typename T>
240bool FramerateController<T>::rateControllerShouldFire(
241 const unsigned int rateControllerId) {
242
243 if (rateControllerId >= rateControllersRegistered) {
244 return true;
245 }
246 T currentTime = static_cast<T>(glfwGetTime());
247 if ((currentTime - rateControllerTimes[rateControllerId]) >
248 rateControllerTargetRates[rateControllerId]) {
249
250 rateControllerTimes[rateControllerId] = currentTime;
251 return true;
252 }
253 return false;
254}
255#endif // ENABLE_RATE_CONTROLLERS
